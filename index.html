CRUD操作（データの追加・取得・更新・削除）
-- C: データを追加（Create）
# INSERT INTO users (name, email) VALUES ('Seto', 'seto@example.com');
-- R: データを取得（Read）
# SELECT * FROM users; # * は全カラムを意味する

WHERE を忘れずに： UPDATE・DELETE に WHERE をつけないと全件が対象になる。必ず条件を指定する。

APIサーバーのコード（index.js）
// 必要なライブラリを読み込む
const express = require('express');
const { Pool } = require('pg');
 
const app = express();
app.use(express.json());
 
// PostgreSQL接続設定
const pool = new Pool({ user: 'postgres', host: 'localhost', database: 'mydb', password: 'postgres', port: 5432 });
 
// GET /users — 全件取得
app.get('/users', async (req, res) => {
  const result = await pool.query('SELECT * FROM users');
  res.json(result.rows);
});
 
// POST /users — データ追加
app.post('/users', async (req, res) => {
  const { name, email } = req.body;
  const result = await pool.query('INSERT INTO users (name, email) VALUES ($1, $2) RETURNING *', [name, email]);
  res.json(result.rows[0]);
});
 
app.listen(3000);

データベース設計（favoritesdb）
-- ジャンルテーブル
genres
  id SERIAL PRIMARY KEY
  name TEXT NOT NULL
-- お気に入りテーブル
favorites
  id SERIAL PRIMARY KEY
  title TEXT NOT NULL
  url TEXT NOT NULL
  genre_id INTEGER REFERENCES genres(id) # 外部キー
REFERENCES（外部キー）： favoritesのgenre_idは必ずgenresのidを参照する。テーブル同士を紐付ける仕組み。
今日の新しいSQL：LEFT JOIN
-- 2つのテーブルを結合してジャンル名も一緒に取得
SELECT f.*, g.name as genre_name
FROM favorites f
LEFT JOIN genres g ON f.genre_id = g.id
ORDER BY f.id DESC;
LEFT JOIN： 左のテーブル（favorites）を基準に、右のテーブル（genres）を結合する。genre_idが一致する行のgenre名を取得できる。
